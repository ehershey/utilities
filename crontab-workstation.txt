MAILTO=cron-workstation-notify@ernie.org
THRESHOLD_DISK_PERCENT=90
NODE_PATH=.

*/5 * * * * if ! pkill -0 Dropbox ; then echo "Dropbox not running on $(hostname)"; fi
*/10 * * * * for percentage in $(df | grep /dev/disk | grep -vi virtualbox | awk '{print $5}' | tr -d \%) ; do if [ "$percentage" -gt $THRESHOLD_DISK_PERCENT ] ; then df | mail -s "disk space problem on $(hostname) ($percentage > $THRESHOLD_DISK_PERCENT)" $MAILTO ; fi ; done
